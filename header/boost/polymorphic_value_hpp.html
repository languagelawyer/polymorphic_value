Header <boost/polymorphic_value.hpp>

namespace boost {
  class bad_polymorphic_value_construction;

  template<typename T> struct default_copy;

  template<typename T> class polymorphic_value;
  template<typename T, typename U = T, class... Ts> 
    polymorphic_value< T > make_polymorphic_value(Ts &&...);
  template<typename T, typename U, typename C = default_copy<U>, 
           typename D = std::default_delete<U> > 
    polymorphic_value< T > assume_polymorphic_value(U *, C = {}, D = {});
  template<typename T, typename U> 
    polymorphic_value< T > 
    polymorphic_value_cast(const polymorphic_value< U > &);
  template<typename T, typename U> 
    polymorphic_value< T > polymorphic_value_cast(polymorphic_value< U > &&);
  template<typename T> 
    void swap(polymorphic_value< T > &, polymorphic_value< T > &);
}
